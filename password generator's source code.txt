    <script>
        function generatePassword() {
            const uppercaseChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
            const lowercaseChars = "abcdefghijklmnopqrstuvwxyz";
            const numberChars = "0123456789";
            const specialChars = "!@#$%^&*";
            
            let passwordChars = "";
            
            if (document.getElementById('includeUppercase').checked) {
                passwordChars += uppercaseChars;
            }
            if (document.getElementById('includeLowercase').checked) {
                passwordChars += lowercaseChars;
            }
            if (document.getElementById('includeNumbers').checked) {
                passwordChars += numberChars;
            }
            if (document.getElementById('includeSpecialChars').checked) {
                passwordChars += specialChars;
            }
            
            let password = "";
            const passwordLength = 12; // You can adjust the length as needed.
            
            for (let i = 0; i < passwordLength; i++) {
                const randomIndex = Math.floor(Math.random() * passwordChars.length);
                password += passwordChars.charAt(randomIndex);
            }
            
            document.getElementById('password').value = password;
        }
    </script>
